public class DivisionLogic {
    // Implements the Euclidean Division Theorem
    public static Result execute(int a, int d) {
        if (d == 0) throw new ArithmeticException();
        
        int q = a / d;
        int r = a % d;

        if (r < 0) {
            r += Math.abs(d);
            q = (a - r) / d;
        }

        return new Result(q, r);
    }
}
