public class EuclideanAlgorithm {
    // Euclidean method to find GCD of two numbers
    public static int findGCD(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }

    // Method to find LCM of two numbers
    public static int findLCM(int a, int b) {
        if (a == 0 || b == 0) {
            return 0;
        }
        int gcd = findGCD(a, b);
        return Math.abs(a * b) / gcd;
    }

    public static void main(String[] args) {
        int n1 = 48, n2 = 18;
        System.out.println("GCD: " + findGCD(n1, n2));
        System.out.println("LCM: " + findLCM(n1, n2));
    }
}
