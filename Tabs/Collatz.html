 * Algorithm: Collatz Conjecture (3n + 1)
 * Stability: STABLE_RELEASE

public class CollatzSequence {
    
    public static void main(String[] args) {
        long n = 27; // Parameter Input
        calculateSequence(n);
    }

    public static void calculateSequence(long n) {
        int steps = 0;
        System.out.println("START_VAL: " + n);

        while (n != 1) {
            if (n % 2 == 0) {
                n = n / 2;
            } else {
                n = (3 * n) + 1;
            }
            steps++;
            System.out.println("STEP_" + steps + ": " + n);
        }
        
        System.out.println("--- CONVERGENCE REACHED ---");
        System.out.println("TOTAL_STEPS: " + steps);
    }
}
